class UGVP_Resonance_v7:

        self.pi = np.pi

        # Объем: Гиперкуб + Сфера
        self.C_vol = self.pi**3 + 4/3.0
        
        # Поверхность: 4D Тор + 2 (Топология границы)
        self.C_surf = 4 * self.pi**2 + 2.0

        self.C_sym = 5 * self.pi**2
        
        # Спаривание
        self.C_pair = self.S_vac / (2 * self.pi)
        
        # Оболочечная поправка (Магический резонанс)
        self.C_shell = self.pi**2 # Сила резонанса

    def calculate_mass(self, element_name, Z, N):
        A = Z + N
        
        # --- 1. ЭНЕРГИЯ СВЯЗИ (ЖИДКАЯ КАПЛЯ) ---
        # Все члены рассчитываются в единицах массы электрона (me)
        
        # Объемная энергия (Glue)
        E_vol = self.C_vol * A
        
        # Поверхностное натяжение (Penalty)
        E_surf = self.C_surf * A**(2/3)
        
        # Кулоновское отталкивание (Strain)
        E_coul = self.C_coul * (Z * (Z - 1)) / (A**(1/3))
        
        # Энергия асимметрии (Isospin Stress)
        E_sym = self.C_sym * ((N - Z)**2) / A
        
        # Энергия спаривания (Spin Coupling)
        if (Z % 2 == 0) and (N % 2 == 0):
            delta = 1.0
        elif (Z % 2 != 0) and (N % 2 != 0):
            delta = -1.0
        else:
            delta = 0.0
        E_pair = self.C_pair * delta / (A**(1/2))
        
        # --- 2. ОБОЛОЧЕЧНАЯ ПОПРАВКА (РЕЗОНАНС) ---
        # Учет магических чисел (2, 8, 20, 28, 50, 82, 126)
        magic_nums = [2, 8, 20, 28, 50, 82, 126]
        
        # Расстояние до ближайшего резонанса
        dN = min([abs(N - m) for m in magic_nums])
        dZ = min([abs(Z - m) for m in magic_nums])
        
        # Гауссово затухание резонанса (Shell Bonus)
        # Геометрия требует экспоненциального подавления при удалении от узла
        E_shell = self.C_shell * (np.exp(-0.5 * dN**2) + np.exp(-0.5 * dZ**2))
        
        # ИТОГОВАЯ ЭНЕРГИЯ СВЯЗИ

        # --- 3. МАССА ЯДРА ---
        M_parts = Z * self.mp_u + N * self.mn_u
        M_final = M_parts - Binding_Energy_u
        
        return M_final, Binding_Energy_u
